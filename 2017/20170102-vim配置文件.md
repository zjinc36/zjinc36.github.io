func! CompileRunGcc()
    exec "w"
    if &filetype == 'c'
        exec '!g++ % -o %<'
        exec '!time ./%<'
    elseif &filetype == 'cpp'
        exec '!g++ % -o %<'
        exec '!time ./%<'
    elseif &filetype == 'python'
        exec '!time python %'
    elseif &filetype == 'sh'
        exec ':!time bash %'
    elseif &filetype == 'java'
        exec "!javac %"
        exec "!java %<"
    endif

"    设置好配置文件后，可通过下述两种方法安装插件:
"    (1) 在Vim尾行模式下运行命令:PluginInstall(旧命令为BundelInstall)

"    至此，需要安装的插件已经安装完毕，可以正常使用了
