#   数据量进制

2.  1byte = 8bit
3.  1kb = 1024byte
4.  1mb = 1024kb
5.  1gb = 1024mb
6.  1tb = 1024gb
7.  1pb = 1024tb
8.  1eb = 1024pb
9.  1zb = 1024eb
